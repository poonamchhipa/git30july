Q.1= swap without third variable 
a=40,b=80

ans= #include <stdio.h>

int main()
{
    int a=40;
    int b=80;
    printf("the value of a is:%d\n",a);
    printf("the value of b is:%d\n",b);
    a=a+b;
    b=a-b;
    a=a-b;
    printf("the swap value of a is:%d\n",a);
     printf("the swap value of b is:%d\n",b);

    return 0;
}
